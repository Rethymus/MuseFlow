abstract class HiveTypeIds {
  static const int fragment = 0;
  static const int appSettings = 1;
  static const int manuscript = 2;
}

class FragmentAdapter extends TypeAdapter<Fragment> {
  @override final int typeId = HiveTypeIds.fragment;
  @override Fragment read(BinaryReader reader) { final json = reader.readMap() as Map<String, dynamic>; return Fragment.fromJson(json); }
  @override void write(BinaryWriter writer, Fragment obj) { writer.writeMap(obj.toJson()); }
}
