class SentenceAiScentScore {
  final String sentence;     // 已 trim 非空
  final int score;           // [0, 100]，越高越像 AI
  final List<String> reasons; // 中文理由
  const SentenceAiScentScore({required this.sentence, required this.score, required this.reasons});
}

class SentenceAiScentAnalysis {
  final List<SentenceAiScentScore> scores; // 按 score 降序
  const SentenceAiScentAnalysis({required this.scores});
  bool get hasNotable => scores.any((s) => s.score >= SentenceAiScentAnalyzer.notableThreshold);
  SentenceAiScentScore? get worst => scores.isEmpty ? null : scores.first;
}

class SentenceAiScentAnalyzer {
  const SentenceAiScentAnalyzer();
  static const int notableThreshold = 30;
  // 句子切分：< 2 CJK 字符的碎片跳过
  SentenceAiScentAnalysis analyze(String text, {int? maxSentences});
}
